/-! ### The GENUINE degree-`≤ n` leaf shape menu (de-vacuification, INSTANTIATION TARGET A)

The two stubs `realClassify := fun _ _ => ⟨[],[]⟩` and `shapesOf := fun _ => ∅` made the capstone
conclusion VACUOUS (empty density sum). `realClassify` is now `B.classify realP n` (above). The menu
below is the genuine `M5.shapeOf`-image of the concrete order-0 leaf OM types of cluster size `0..n`
(each a real `M5.OMType` via `leafOMTypeOfSize`), so it is a NON-EMPTY `Finset` of REAL degree-`≤ n`
shapes — discharging `shapesOf_real`/`finiteTermination` while making the density sum non-empty. -/

/-- The finite list of genuine order-0 leaf OM types of cluster size `0..n` (`leafOMTypeOfSize s`,
each a real `M5.OMType`). -/
noncomputable def leafTypesUpTo (n : ℕ) : List M5.OMType :=
  (List.range (n + 1)).map leafOMTypeOfSize

/-- **The genuine degree-`≤ n` leaf shape menu**: the `M5.shapeOf`-image of `leafTypesUpTo n`. A
NON-EMPTY `Finset ClusterShape` of REAL shapes (finiteness automatic by `Finset`). -/
noncomputable def shapeMenu (n : ℕ) : Finset ClusterShape :=
  ((leafTypesUpTo n).map M5.shapeOf).toFinset

/-- Every shape in `shapeMenu n` is `M5.shapeOf` of a leaf OM type of cluster size `≤ n`. The single
place where the menu's REALITY is consumed (by `shapesOf_real`/`finiteTermination`). -/
theorem shapeMenu_mem_real (n : ℕ) : ∀ T ∈ shapeMenu n,
    ∃ T' : M5.OMType, T = M5.shapeOf T' ∧ M5.clusterSize T' ≤ n := by
  intro T hT
  -- `T ∈ ((List.range (n+1)).map leafOMTypeOfSize).map M5.shapeOf).toFinset` ⇒ `T = shapeOf (leaf s)`,
  -- some `s < n+1`. Peel the two `List.map`s and the `List.range`.
  simp only [shapeMenu, leafTypesUpTo, List.mem_toFinset, List.mem_map, List.mem_range] at hT
  obtain ⟨T', hT', hTeq⟩ := hT
  obtain ⟨s, hs, hs'⟩ := hT'
  refine ⟨leafOMTypeOfSize s, ?_, ?_⟩
  · rw [← hTeq, ← hs']
  · show M5.clusterSize (leafOMTypeOfSize s) ≤ n
    -- `clusterSize T' = T'.cs` and `(leafOMTypeOfSize s).cs = s`, both by `rfl`; `s < n+1 ⇒ s ≤ n`.
    have : M5.clusterSize (leafOMTypeOfSize s) = s := rfl
    omega

/-- The menu is NON-EMPTY (it contains the size-0 leaf shape). This is the witness that
de-vacuifies the density sum `∑_{T ∈ shapeMenu n} coeff T q`. -/
theorem shapeMenu_nonempty (n : ℕ) : (shapeMenu n).Nonempty := by
  refine ⟨M5.shapeOf (leafOMTypeOfSize 0), ?_⟩
  simp only [shapeMenu, leafTypesUpTo, List.mem_toFinset, List.mem_map, List.mem_range]
  exact ⟨leafOMTypeOfSize 0, ⟨0, Nat.zero_lt_succ n, rfl⟩, rfl⟩

/-- **The shape menu of a type `σ`** — σ-KEYED to the capstone's distinguished type `σ₀`. For `σ = σ₀`
it is the genuine non-empty `shapeMenu n`; off `σ₀` it is empty. This σ-keying is REQUIRED to keep
`CountingModel.decidedCount_off_menu` (vanishing off the `typeMenu = {σ₀}`) consistent with a genuinely
non-zero decided count at `σ₀` (see `realM.box_partition`). -/
noncomputable def shapesOf (σ₀ : FactorizationType) : FactorizationType → Finset ClusterShape :=
  fun σ => if σ = σ₀ then shapeMenu n else ∅

/-- The raw per-type decided count `#{ f ∈ monicBox(p,N,n) decided to type σ }`, TIED to the stratum
partition: it IS the sum of the genuine shape-fiber counts over the shape menu `shapesOf σ₀ σ`. So the
M7 `partition` identity holds by construction (`rfl`). -/
noncomputable def rawDecided (σ₀ : FactorizationType) : FactorizationType → ℕ → ℕ :=
  fun σ N => ∑ T ∈ shapesOf n σ₀ σ, rawCount n T N

/-- **The shape menu lists only REAL shapes of degree-`≤ n` OM types** (the `finiteTermination`
coupling), NOW NON-VACUOUS. For `σ.degree = n`, every `T ∈ shapesOf σ₀ σ` is `M5.shapeOf` of an OM
type of cluster size `≤ n`: either `σ = σ₀` (so the menu is `shapeMenu n`, discharged by
`shapeMenu_mem_real`) or `σ ≠ σ₀` (so the menu is `∅`, membership vacuous). -/
theorem shapesOf_real (σ₀ : FactorizationType) : ∀ (σ : FactorizationType), σ.degree = n →
    ∀ T ∈ shapesOf n σ₀ σ, ∃ T' : M5.OMType, T = M5.shapeOf T' ∧ M5.clusterSize T' ≤ n := by
  intro σ _ T hT
  simp only [shapesOf] at hT
  by_cases hσ : σ = σ₀
  · rw [if_pos hσ] at hT
    exact shapeMenu_mem_real n T hT
  · rw [if_neg hσ] at hT
    exact absurd hT (Finset.notMem_empty T)

  `≥ 0`, the pivot `1 - q^{-w} ∈ [0,1]`).
* `h_decided_le_box` — disjoint-fiber counting: `∑_{T ∈ shapeMenu} stratumCount T N ≤ Nat.card box =

/-- The per-cell count coefficient `(q^δ)^{dS-1} · bb1Value polygon q` is nonnegative. Pure sign fact
(each of the three factors is `≥ 0`; for `(1-q⁻¹)^V` use `q⁻¹ ≤ 1` for `q ≥ 1`, and `q = 0` gives
`(1-0)^V`). `[core]`. -/
theorem countCellCoeff_nonneg (c : CountCell) (q' : ℕ) : 0 ≤ countCellCoeff c q' := by
  unfold countCellCoeff L4.bb1Value
  have h2 : (0:ℚ) ≤ (1 - (q':ℚ)⁻¹) ^ L4.newtonVertexCount c.polygon := by
    apply pow_nonneg
    rcases Nat.eq_zero_or_pos q' with h | h
    · simp [h]
    · have h1 : (1:ℚ) ≤ (q':ℚ) := by exact_mod_cast h
      have : (q':ℚ)⁻¹ ≤ 1 := by rw [inv_le_one_iff₀]; right; exact h1
      linarith
  positivity

/-- The self-loop pivot `countPivot s q'` is nonnegative: `1` at leaves, `1 - (q'^w)⁻¹ ∈ [0,1]` at
ramifying nodes (using `(q'^w)⁻¹ ≤ 1`). `[core]`. -/
theorem countPivot_nonneg (s q' : ℕ) : 0 ≤ countPivot s q' := by
  unfold countPivot
  by_cases h : 2 ≤ s
  · simp only [h, if_true]
    have hle : ((q':ℚ)^L5fix.selfLoopExponent s)⁻¹ ≤ 1 := by
      rcases Nat.eq_zero_or_pos q' with hz | hz
      · rcases Nat.eq_zero_or_pos (L5fix.selfLoopExponent s) with hw | hw
        · simp [hz, hw]
        · rw [hz]; push_cast; rw [zero_pow (by omega)]; simp
      · have h1 : (1:ℚ) ≤ (q':ℚ) := by exact_mod_cast hz
        rw [inv_le_one_iff₀]; right; exact one_le_pow₀ h1
    linarith
  · simp only [h, if_false]; norm_num

/-- **`clusterCount` is nonnegative** — the elementary sign fact discharging `h_coeff_nonneg`. By
well-founded induction on `treeSize`: the recursion is a `div` of a `List.sum` of products of
`countCellCoeff` (`≥ 0`) and child `clusterCount`s (IH) by `countPivot` (`≥ 0`). Holds for ALL `q' : ℕ`
(no `1 < q'` needed). `[core]`. -/
theorem clusterCount_nonneg
    (cells : ClusterShape → List CountCell) (treeSize : ClusterShape → ℕ)
    (hdesc : ∀ (T : ClusterShape), ∀ c ∈ cells T, ∀ ch ∈ c.children, treeSize ch < treeSize T)
    (T : ClusterShape) (q' : ℕ) :
    0 ≤ clusterCount cells treeSize hdesc T q' := by
  induction T using (measure treeSize).wf.induction with
  | _ T ih =>
    rw [clusterCount_rec cells treeSize hdesc T q']
    apply div_nonneg _ (countPivot_nonneg _ _)
    apply List.sum_nonneg
    intro x hx; simp only [List.mem_map] at hx
    obtain ⟨c, hc, rfl⟩ := hx
    apply mul_nonneg (countCellCoeff_nonneg c q')
    apply List.prod_nonneg
    intro y hy; simp only [List.mem_map] at hy
    obtain ⟨ch, hch, rfl⟩ := hy
    exact ih ch (hdesc T c hc ch hch)

/-- **Disjoint-fiber sum bound** (generic, `[core]`). The fibers `{f // classify f = T}` are disjoint
subsets of `α` indexed by distinct `T`, so the sum of their cardinalities over any finite set of shapes
is `≤ Nat.card α`. Proved via `Finset.card_biUnion` of the disjoint filtered fibers `⊆ univ`. -/
theorem fiber_sum_le_card {α β : Type*} [Finite α] [DecidableEq β]
    (classify : α → β) (S : Finset β) :
    (∑ T ∈ S, Nat.card {f : α // classify f = T}) ≤ Nat.card α := by
  classical
  haveI : Fintype α := Fintype.ofFinite α
  have hrw : ∀ T : β, Nat.card {f : α // classify f = T}
      = (Finset.univ.filter (fun f => classify f = T)).card := by
    intro T; rw [Nat.card_eq_fintype_card, Fintype.card_subtype]
  simp_rw [hrw]; rw [Nat.card_eq_fintype_card]
  calc (∑ T ∈ S, (Finset.univ.filter (fun f => classify f = T)).card)
      = (S.biUnion (fun T => Finset.univ.filter (fun f => classify f = T))).card := by
        rw [Finset.card_biUnion]
        intro x _ y _ hxy
        simp only [Finset.disjoint_left, Finset.mem_filter]
        rintro a ⟨_, rfl⟩ ⟨_, h2⟩; exact hxy h2
    _ ≤ (Finset.univ : Finset α).card := Finset.card_le_card (Finset.subset_univ _)
    _ = Fintype.card α := Finset.card_univ

/-- **`h_decided_le_box`, PROVED for the real instance.** The decided stratum mass over the shape menu
is `≤ q^{nN}`. The summands are the cast `Nat.card` shape fibers over `monicBox(realP=2,N,n)`; they are
disjoint (single-valued classifier `realClassify n N`), so `∑ ≤ Nat.card box`. For `0 < N`,
`Nat.card box = (2^N)^n ≤ q^{nN}` since `2 = realP ≤ q`; at `N = 0` the box over `ZMod (2^0)=ZMod 1` is a
subsingleton (`Nat.card ≤ 1 = q^0`). `[core]`. -/
theorem decided_le_box (hq : 2 ≤ q) (N : ℕ) :
    (∑ T ∈ shapeMenu n, M8.stratumCount (rawCount n) T N) ≤ (q : ℚ) ^ (n * N) := by
  have hfib : (∑ T ∈ shapeMenu n, M8.stratumCount (rawCount n) T N)
      = ((∑ T ∈ shapeMenu n,
          Nat.card {f : QuotientBox.monicBox realP N n // realClassify n N f = T} : ℕ) : ℚ) := by
    push_cast
    rfl
  rw [hfib]
  have hle : (∑ T ∈ shapeMenu n,
        Nat.card {f : QuotientBox.monicBox realP N n // realClassify n N f = T})
      ≤ Nat.card (QuotientBox.monicBox realP N n) :=
    fiber_sum_le_card (realClassify n N) (shapeMenu n)
  have hcast : ((∑ T ∈ shapeMenu n,
        Nat.card {f : QuotientBox.monicBox realP N n // realClassify n N f = T} : ℕ) : ℚ)
      ≤ (Nat.card (QuotientBox.monicBox realP N n) : ℚ) := by exact_mod_cast hle
  refine le_trans hcast ?_
  rcases Nat.eq_zero_or_pos N with hN | hN
  · subst hN
    have hsub : Subsingleton (QuotientBox.monicBox realP 0 n) := by
      have : Subsingleton (QuotientBox.ResidueRing realP 0) := by
        show Subsingleton (ZMod (realP ^ 0))
        rw [pow_zero]; infer_instance
      constructor
      intro a b
      exact Subtype.ext (Polynomial.ext (fun i => Subsingleton.elim _ _))
    have hcard1 : Nat.card (QuotientBox.monicBox realP 0 n) ≤ 1 :=
      Finite.card_le_one_iff_subsingleton.mpr hsub
    have : (Nat.card (QuotientBox.monicBox realP 0 n) : ℚ) ≤ 1 := by exact_mod_cast hcard1
    simpa using this
  · rw [QuotientBox.card_monicBox realP N n hN]
    push_cast
    have h2q : (realP : ℚ) ^ N ≤ (q : ℚ) ^ N := by
      apply pow_le_pow_left₀ (by norm_num)
      show (realP : ℚ) ≤ (q : ℚ)
      have hrq : realP ≤ q := by unfold realP; omega
      exact_mod_cast hrq
    calc ((realP : ℚ) ^ N) ^ n ≤ ((q : ℚ) ^ N) ^ n := by
          apply pow_le_pow_left₀ (by positivity) h2q
      _ = (q : ℚ) ^ (n * N) := by rw [← pow_mul, Nat.mul_comm]

/-! ## The per-stratum count limit (PORTED from `MontesAxiom.nodeMultiplicativity`)

              countCellCoeff c q

                    (M7.cellsOfShapeWF_descend) ch q)).prod)).sum
            / countPivot (M6.treeSize T) q)))
    (T : ClusterShape) :
    Filter.Tendsto (fun N => M8.stratumCount (rawCount n) T N / (q : ℚ) ^ (n * N))
      Filter.atTop (nhds (clusterCount (M7.cellsOfShapeWF) M6.treeSize
        (M7.cellsOfShapeWF_descend) T q)) := by

  have hbox := h_node T
  -- `rawCount n = M8.shapeFiberCount realP n (realClassify n)`, so the LHS matches `hbox` definitionally.
  -- recombine the box-additive sum (per-box = box-volume × residual-count) into `clusterCount`:

      (((M7.cellsOfShapeWF T).map (fun c =>
            countCellCoeff c q
              * (c.children.map (fun ch => clusterCount (M7.cellsOfShapeWF) M6.treeSize
                  (M7.cellsOfShapeWF_descend) ch q)).prod)).sum
          / countPivot (M6.treeSize T) q)
        = clusterCount (M7.cellsOfShapeWF) M6.treeSize
            (M7.cellsOfShapeWF_descend) T q := by

        * (c.children.map (fun ch => clusterCount (M7.cellsOfShapeWF) M6.treeSize
            (M7.cellsOfShapeWF_descend) ch q)).prod)
      (fun c _ => rfl)]
  rwa [hrecomb] at hbox
